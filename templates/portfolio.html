SoftwareSerial bluetooth(2, 3);

void setup() {
  Serial.begin(9600);
  bluetooth.begin(9600);
  pinMode(motorL1, OUTPUT);
  pinMode(motorL2, OUTPUT);
  pinMode(motorR1, OUTPUT);
  pinMode(motorR2, OUTPUT);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
}

void loop() {
  long distance = getDistance();
  
  if (bluetooth.available()) {
    String cmd = bluetooth.readString();
    processCommand(cmd);
  }
  
  if (distance > 20) {
    moveForward();
  } else {
    stopMotors();
    avoidObstacle();
  }
  delay(50);
}

long getDistance() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  
  long duration = pulseIn(echoPin, HIGH);
  return duration * 0.034 / 2;
}

AI Assistant Main
import speech_recognition as sr
import pyttsx3
import requests
from datetime import datetime

class HospitalAssistant:
    def __init__(self):
        self.recognizer = sr.Recognizer()
        self.tts = pyttsx3.init()
        self.patient_data = {}
    
    def greet_patient(self):
        greeting = "Hello! Please state your name."
        self.speak(greeting)
        return self.listen()
    
    def speak(self, text):
        self.tts.say(text)
        self.tts.runAndWait()
    
    def listen(self):
        with sr.Microphone() as source:
            audio = self.recognizer.listen(source)
            try:
                response = self.recognizer.recognize_google(audio)
                return response
            except:
                return None
    
    def collect_info(self):
        name = self.get_name()
        symptoms = self.get_symptoms()
        
        self.patient_data = {
            'name': name,
            'symptoms': symptoms,
            'time': datetime.now()
        }
        
        return self.register_patient()
    
    def register_patient(self):
        try:
            response = requests.post(
                "http://hospital-api/register",
                json=self.patient_data
            )
            
            if response.status_code == 200:
                queue_num = response.json()['queue']
                self.speak(f"Queue number: {queue_num}")
                return True
            return False
        except:
            return False

assistant = HospitalAssistant()
assistant.run()

#include <WiFi.h>
#include <WebSocketsServer.h>

const char* ssid = "YourWiFi";
const char* password = "YourPass";

WebSocketsServer webSocket = WebSocketsServer(81);

const int motorL1 = 18, motorL2 = 19;
const int motorR1 = 21, motorR2 = 22;

struct RobotState {
  float targetX = 0;
  float targetY = 0;
  bool isTracking = false;
} state;

void setup() {
  Serial.begin(115200);
  
  pinMode(motorL1, OUTPUT);
  pinMode(motorL2, OUTPUT);
  pinMode(motorR1, OUTPUT);
  pinMode(motorR2, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
  }
  
  webSocket.begin();
  webSocket.onEvent(webSocketEvent);
}

void loop() {
  webSocket.loop();
  
  if (state.isTracking) {
    trackTarget();
  }
  
  delay(50);
}

void trackTarget() {
  float centerX = 160;
  float centerY = 120;
  
  float errorX = state.targetX - centerX;
  float errorY = state.targetY - centerY;
  
  if (abs(errorX) < 20 && abs(errorY) < 20) {
    stopRobot();
    return;
  }
  
  int speed = 150;
  int turn = map(abs(errorX), 0, 160, 0, 100);
  
  if (errorX > 20) {
    moveRobot(speed + turn, speed - turn);
  } else if (errorX < -20) {
    moveRobot(speed - turn, speed + turn);
  }
}

Battery Management System
#include <Arduino.h>

const int solarPin = A0;
const int batteryPin = A1;
const int motorPin = 9;
const int ledPin = 13;

float batteryVoltage = 0;
float solarVoltage = 0;
bool solarCharging = false;

void setup() {
  Serial.begin(9600);
  pinMode(motorPin, OUTPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  readSensors();
  managePower();
  controlMotor();
  
  displayStatus();
  delay(1000);
}

void readSensors() {
  solarVoltage = analogRead(solarPin) * (5.0 / 1023.0);
  batteryVoltage = analogRead(batteryPin) * (5.0 / 1023.0);
}

void managePower() {
  if (solarVoltage > 4.0 && batteryVoltage < 4.0) {
    solarCharging = true;
    digitalWrite(ledPin, HIGH);
  } else {
    solarCharging = false;
    digitalWrite(ledPin, LOW);
  }
}

void controlMotor() {
  if (batteryVoltage > 3.2) {
    int motorSpeed = map(batteryVoltage * 100, 320, 420, 100, 255);
    analogWrite(motorPin, motorSpeed);
  } else {
    analogWrite(motorPin, 0); // Low battery protection
  }
}

void displayStatus() {
  Serial.print("Battery: ");
  Serial.print(batteryVoltage);
  Serial.print("V, Solar: ");
  Serial.print(solarVoltage);
  Serial.print("V, Charging: ");
  Serial.println(solarCharging ? "Yes" : "No");
}
